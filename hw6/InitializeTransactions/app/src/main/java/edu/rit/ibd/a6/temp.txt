SELECT tid, iid FROM Actor JOIN Movie ON id=mid WHERE year BETWEEN 1907 AND 2021 AND totalvotes > 20 AND mid IN (SELECT mid FROM MovieGenre JOIN Genre ON id=gid WHERE name LIKE "Thriller") UNION SELECT tid, iid FROM Director JOIN Movie ON id=mid WHERE year BETWEEN 1907 AND 2021 AND totalvotes > 20 AND mid IN (SELECT mid FROM MovieGenre JOIN Genre ON id=gid WHERE name LIKE Thriller) UNION SELECT tid, iid FROM Producer JOIN Movie ON id=mid WHERE year BETWEEN 1907 AND 2021 AND totalvotes > 20 AND mid IN (SELECT mid FROM MovieGenre JOIN Genre ON id=gid WHERE name LIKE "Thriller") UNION SELECT tid, iid FROM Writer JOIN Movie ON id=mid WHERE year BETWEEN 1907 AND 2021 AND totalvotes > 20 AND mid IN (SELECT mid FROM MovieGenre JOIN Genre ON id=gid WHERE name LIKE "Thriller") ORDER BY tid, iid;
